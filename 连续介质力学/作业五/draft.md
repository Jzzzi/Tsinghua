=(\frac {\partial (v^i \vec e_i)}{\partial t})_{\xi^\alpha}
\\
=(\frac{\partial v^i}{\partial t})_{\xi^\alpha} \vec e_i+v^i (\frac{\partial \vec e_i}{\partial t})_{\xi^\alpha}
